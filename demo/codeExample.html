package cn.itcast_01;
          /*
           * 最简单的枚举，也是用的最多的
           */
          public enum Direction1 {
            FONT, BEHIND, LEFT, RIGHT;
          }
        
          package cn.itcast_01;

          public enum Direction2 {

            FONT("前"), BEHIND("后"), LEFT("左"), RIGHT("右");
            
            private String str;

            private Direction2(String str) {
              this.str = str;
            }
            
            public String getStr(){
              return str;
            }
            
          }
        
          package cn.itcast_01;

          public enum Direction3 {

            FONT("前"){
              @Override
              public void show() {
                
                System.out.println("qian");
              }
            }, 
            BEHIND("后"){
              @Override
              public void show() {
                
                System.out.println("hou");
              }
            }, 
            LEFT("左"){
              @Override
              public void show() {
                
                System.out.println("zuo");
              }
            }, 
            RIGHT("右"){
              @Override
              public void show() {
                
                System.out.println("you");
              }
            };
            
            private String str;

            private Direction3(String str) {
              this.str = str;
            }
            
            public String getStr(){
              return str;
            }
            
            public abstract void show();
            
          }

        
          package cn.itcast_01;

          import javax.naming.directory.DirContext;

          public class DirectionTest {

            public static void main(String[] args) {
              Direction1 d1 = Direction1.FONT;
              System.out.println(d1);
              System.out.println("------");
              
              Direction2 d2 = Direction2.FONT;
              System.out.println(d2);
              System.out.println(d2.getStr());
              System.out.println("------");
              
              Direction3 d3 = Direction3.FONT;
              System.out.println(d3);
              System.out.println(d3.getStr());
              d3.show();
            }
          }
